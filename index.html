⚙️ Mesures DAX – Référentiel Complet Power BI
📌 Segments (Slicers) recommandés pour votre rapport : 📅 Année 📂 Type de dette 📊 Indicateur 📏 Seuil / Norme 🔔 Statut
📁Dossier 01 – KPI Global
📐 Dim_Date (Table Calendrier)
Dim_Date =
ADDCOLUMNS(
    CALENDAR(DATE(2020,1,1), DATE(2035,12,31)),
    "Année",    YEAR([Date]),
    "Mois",     FORMAT([Date], "MMMM"),
    "NumMois",  MONTH([Date]),
    "Trimestre","T" & FORMAT([Date], "Q")
)
💰 Dette_Totale
Dette_Totale = SUM(AVD[Dette Totale])
🏭 PIB_Nominal
PIB_Nominal = SUM(AVD[PIB Nominal])
📊 Ratio_Dette_PIB
Ratio_Dette_PIB =
DIVIDE([Dette_Totale], [PIB_Nominal], 0) * 100
📏 Norme_UEMOA
Norme_UEMOA = AVERAGE(AVD[Norme UEMOA])
⚡ Ecart_Norme_UEMOA
Ecart_Norme_UEMOA =
[Ratio_Dette_PIB] - [Norme_UEMOA]
📁Dossier 02 – Structure de la Dette
🌍 Dette_Exterieure
Dette_Exterieure = SUM(AVD[Dette Extérieure])
🏠 Dette_Interieure
Dette_Interieure = SUM(AVD[Dette Intérieure])
📊 Ratio_Dette_Ext_PIB
Ratio_Dette_Ext_PIB =
DIVIDE([Dette_Exterieure], [PIB_Nominal], 0) * 100
📊 Ratio_Dette_Int_PIB
Ratio_Dette_Int_PIB =
DIVIDE([Dette_Interieure], [PIB_Nominal], 0) * 100
🔢 Part_Dette_Exterieure
Part_Dette_Exterieure =
DIVIDE([Dette_Exterieure], [Dette_Totale], 0) * 100
🔢 Part_Dette_Interieure
Part_Dette_Interieure =
DIVIDE([Dette_Interieure], [Dette_Totale], 0) * 100
🗂️ Colonne calculée – Etat_AVD
Etat_AVD =
IF(
    AVD[Dette % PIB] <= AVD[Norme UEMOA],
    "Sous norme",
    "Au-dessus norme"
)
🗂️ Colonne calculée – Categorie_Risque
Categorie_Risque =
SWITCH(
    TRUE(),
    AVD[Dette % PIB] < 50, "Faible",
    AVD[Dette % PIB] < 70, "Modéré",
    "Élevé"
)
📁Dossier 03 – Emprunts-Programmes
📋 Emprunts_Programmes
Emprunts_Programmes = SUM(AVD[Emprunts-programmes])
⚖️ Poids_Emprunts_Programmes
Poids_Emprunts_Programmes =
DIVIDE([Emprunts_Programmes], [Dette_Totale], 0) * 100
📁Dossier 04 – Mesures d'Évolution
📅 Dette_Totale_Annee_Precedente
Dette_Totale_Annee_Precedente =
CALCULATE(
    [Dette_Totale],
    DATEADD(Dim_Date[Date], -1, YEAR)
)
📈 Variation_Dette_Valeur
Variation_Dette_Valeur =
[Dette_Totale] - [Dette_Totale_Annee_Precedente]
📊 Croissance_Dette
Croissance_Dette =
DIVIDE(
    [Variation_Dette_Valeur],
    [Dette_Totale_Annee_Precedente],
    0
) * 100
📉 Ratio_Dette_PIB_Annee_Precedente
Ratio_Dette_PIB_Annee_Precedente =
CALCULATE(
    [Ratio_Dette_PIB],
    DATEADD(Dim_Date[Date], -1, YEAR)
)
📊 Variation_Ratio_Dette_PIB
Variation_Ratio_Dette_PIB =
[Ratio_Dette_PIB] - [Ratio_Dette_PIB_Annee_Precedente]
📁Dossier 05 – Alertes & Conformité
📏 Seuil_AVD
Seuil_AVD = AVERAGE(AVD[Seuil])
✅ Statut_Conformite
Statut_Conformite =
IF(
    [Ratio_Dette_PIB] <= [Norme_UEMOA],
    "Conforme",
    "Non conforme"
)
🚨 Alerte_AVD
Alerte_AVD =
IF(
    [Ratio_Dette_PIB] > [Seuil_AVD],
    "Alerte rouge",
    "Situation maîtrisée"
)
